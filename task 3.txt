def find_largest():
    a = int(input("Enter the first number: "))
    b = int(input("Enter the second number: "))
    c = int(input("Enter the third number: "))

    nums = {"A": a, "B": b, "C": c}

    largest_letter = max(nums, key=nums.get)
    largest_value = nums[largest_letter]

    print(f"Letter {largest_letter} is the largest with a value of {largest_value}.")

find_largest()
